* ============================================================
* Atlantic salmon growth model
* Based on: Handeland et al. (2008) Aquaculture 283:36-42
*           Bjornsson & Steinarsson (2002); Jobling (1994)
*
* Model: SGR = f(T, W) — bell-shaped in T, declining in W
*   SGR(%/day) = alpha * W^beta * [a*T + b*T^2]
*
* Parameterisation anchored to Handeland 2008 empirical values:
*   At W~150g: SGR(6°C)=0.78, SGR(10°C)=1.35, SGR(14°C)=1.53, SGR(18°C)=1.29
*   Optimum T_opt ~ 13-14°C for small fish, shifts slightly with size
*   SGR declines with weight: beta ~ -0.30 (allometric exponent)
*
* Integration: daily Euler method
*   W(t+1) = W(t) * exp(SGR(T,W(t))/100)
* ============================================================

clear
set obs 401
gen day = _n - 1

* --- Parameters ---
local W0    = 50        // initial weight (g)
local beta  = -0.30     // allometric size exponent (SGR declines with W)
local Topt_base = 13.5  // optimum temperature at reference size (°C)
local k     = 0.018     // curvature of bell curve (controls width)

* Calibrated scaling so SGR at W=150, T=14 ≈ 1.53 %/day
* SGR(T,W) = alpha * W^beta * exp(-k*(T - Topt)^2)
* alpha calibrated: 1.53 = alpha * 150^(-0.30) * exp(0)
* => alpha = 1.53 / 150^(-0.30)
local alpha = 1.53 / (150^(-0.30))

* --- Simulate growth for each temperature ---
foreach temp of numlist 4(2)20 {

    gen w_temp`temp' = `W0'
    
    forvalues d = 1/400 {
        local prev = d - 1
        * SGR bell curve in temperature
        quietly replace w_temp`temp' = ///
            w_temp`temp'[_n-1] * ///
            exp( (`alpha' * (w_temp`temp'[_n-1])^(`beta') * ///
                  exp(-`k' * (`temp' - `Topt_base')^2)) / 100 ) ///
            if day == `d'
    }
}

* --- Plot ---
twoway ///
    (line w_temp4  day, lcolor(navy)    lwidth(medthin))  ///
    (line w_temp6  day, lcolor(blue)    lwidth(medthin))  ///
    (line w_temp8  day, lcolor(dkgreen) lwidth(medthin))  ///
    (line w_temp10 day, lcolor(green)   lwidth(medium))   ///
    (line w_temp12 day, lcolor(lime)    lwidth(medium))   ///
    (line w_temp14 day, lcolor(orange)  lwidth(thick) lpattern(solid)) ///
    (line w_temp16 day, lcolor(red)     lwidth(medium))   ///
    (line w_temp18 day, lcolor(cranberry) lwidth(medthin)) ///
    (line w_temp20 day, lcolor(maroon)  lwidth(medthin))  ///
    , ///
    ytitle("Weight (g)") ///
    xtitle("Day post sea-transfer") ///
    title("Atlantic salmon growth by temperature") ///
    subtitle("SGR model: Handeland et al. (2008), bell-shaped T response") ///
    note("Optimum T ~13-14°C; SGR declines with weight (W^-0.30)." ///
         "18-20°C growth ≈ 10°C due to elevated maintenance metabolism.") ///
    legend(order( ///
        1 "4°C"  2 "6°C"  3 "8°C"  4 "10°C" 5 "12°C" ///
        6 "14°C (optimum)" 7 "16°C" 8 "18°C" 9 "20°C") ///
        cols(3) size(small)) ///
    graphregion(color(white))
